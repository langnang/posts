## 数据模板定义

### String

`'name|min-max': string`

```js
Mock.mock({
  "string|1-10": "★"
})
```

```js
{
  "string": "★★★★★★★★★★"
}
```

'name|count': string

```js
Mock.mock({
  "string|3": "★★★"
})
```

```js
{
  "string": "★★★★★★★★★"
}
```

### Path

```js
Mock.mock({
  "foo": "Hello",
  "nested": {
    "a": {
      "b": {
        "c": "Mock.js"
      }
    }
  },
  "absolutePath": "@/foo @/nested/a/b/c"
})
```

```js
{
  "foo": "Hello",
  "nested": {
    "a": {
      "b": {
        "c": "Mock.js"
      }
    }
  },
  "absolutePath": "Hello Mock.js"
}
```

Relative Path

```js

Mock.mock({
  "foo": "Hello",
  "nested": {
    "a": {
      "b": {
        "c": "Mock.js"
      }
    }
  },
  "relativePath": {
    "a": {
      "b": {
        "c": "@../../../foo @../../../nested/a/b/c"
      }
    }
  }
})

```
